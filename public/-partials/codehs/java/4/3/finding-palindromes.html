4.3.8 Finding Palindromes

import java.util.Scanner; public class Palindromes { /** * This program lets the user input some text and * prints out whether or not that text is a palindrome. */ public static void main(String[] args) { Scanner sc = new Scanner(System.in); String word = sc.nextLine(); System.out.println("Your word is " + (isPalindrome(word)?"":"not ") + "a palindrome!"); } /** * This method determines if a String is a palindrome, * which means it is the same forwards and backwards. * * @param text The text we want to determine if it is a palindrome. * @return A boolean of whether or not it was a palindrome. */ public static boolean isPalindrome(String text) { return text.equals(reverse(text)); } /** * This method reverses a String. * * @param text The string to reverse. * @return The new reversed String. */ public static String reverse(String text) { String r = ""; for(int i = text.length() - 1; i>=0;i--) r+= text.charAt(i); return r; } }
