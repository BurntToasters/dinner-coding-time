public class MultiplicationTable { public static void main(String[] args) { // Call makeMultiplicationTable makeMultiplicationTable(); } // Makes a multiplcation table public static void makeMultiplicationTable() { for(int i = 1; i <= 10; i++) { for(int j = 1; j <= 10; j++) System.out.print(i*j + "\t"); System.out.println(); } } }
