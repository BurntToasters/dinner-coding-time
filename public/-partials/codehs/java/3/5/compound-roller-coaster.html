3.5.7 Compound Roller Coaster

import java.util.Scanner; public class RollerCoaster { public static void main(String[] args) { Scanner input = new Scanner(System.in); System.out.println("Enter Height (inches):"); int height = input.nextInt(); System.out.println("Enter Age (years):"); int age = input.nextInt(); if(age > 8 && height > 41) System.out.println("Welcome aboard!"); else System.out.println("Sorry, you are not eligible to ride"); } }
