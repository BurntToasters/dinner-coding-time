import java.util.ArrayList; public class CardCollection { private ArrayList<Card> collection; public CardCollection(ArrayList<Card> theCollection) { this.collection = theCollection; } public double totalValue() { double total = 0.0; for(int i = 0; i < collection.size(); i++) { total += collection.get(i).getValue(); } return total; } public void checkPerfect() { for(int i = 0; i < collection.size(); i++) { if(collection.get(i).getCondition().equals("perfect")) System.out.println(collection.get(i).getName()); } } public void orderNumerically() { //horribly inoptimal sort algo, but we haven't learned anything better for(int i = 0; i < collection.size(); i++) { int setNumBase = collection.get(i).getSetNum(); for(int j = i + 1; j < collection.size(); j++) { Card toMove = collection.get(j); if(toMove.getSetNum() < setNumBase) { collection.remove(j); collection.add(i, toMove); j--; } } } } }
