Word Count

import java.util.Scanner; public class WordCount { public static void main(String[] args) { Scanner sc = new Scanner(System.in); System.out.println("Give me a string of text"); String text = sc.nextLine(); System.out.println("Give me a single character"); //this is weird but char doesn't have its own method. this is how you'd do it instead char letter = sc.next().charAt(0); System.out.println("Word count: " + wordCount(text)); System.out.println("Percent of words that start with letter: " + (percentStartWith(text, letter)*100) + "%"); } public static int wordCount(String text) { //it'll always have at least 1 word int words = 1; for(int i = 0; i < text.length(); i++) { if(text.charAt(i) == ' ') { words++; } } return words; } public static double percentStartWith(String text, char letter) { int totalWordsCount = wordCount(text); int wordsThatStartWith = 0; //the first letter will always start a word, so this starts as `true` boolean nextCharacterStartsWord = true; for(int i = 0; i < text.length(); i++) { //if this is the start of the word, process it if(nextCharacterStartsWord) { if(text.charAt(i) == letter) { wordsThatStartWith++; } } //when this character is a space, remember that the next character will be the start of a word; //this will be checked in the next iteration. //reset it to false when this character *isn't* a space if(text.charAt(i) == ' ') { nextCharacterStartsWord = true; } else { nextCharacterStartsWord = false; } } return wordsThatStartWith / ((double)totalWordsCount); } }
